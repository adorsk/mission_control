        self.items = [
            {'prop_{i}'.format(i=i): 'item_{idx}_val_{i}'.format(idx=idx, i=i)
             for i in range(3)}
            for idx in range(3)

                'prop_0': value['prop_0'],
                'prop_1': 'new_prop_1_val',
                'prop_2': 'ctx_2_val',
                'prop_3': 'skel_3_val',
            }
            for idx, (key, value) in enumerate(enumerate(self.items))
        ]
